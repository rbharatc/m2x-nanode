548 int M2XNanodeClient::getTimestampSeconds(int32_t* ts) {
549  // The maximum value of signed 64-bit integer is 0x7fffffffffffffff,
550  // which is 9223372036854775807. It consists of 19 characters, so a
551  // buffer of 20 is definitely enough here
552  int length = 20;
553  char buffer[20];
554  int status = getTimestamp(buffer, &length, 1);
555  if (status == 200) {
556  int32_t result = 0;
557  for (int i = 0; i < length; i++) {
558  result = result * 10 + (buffer[i] - '0');
559  }

561  }
562  return status;

565 // Encodes and prints string using Percent-encoding specified
566 // in RFC 1738, Section 2.2
567 int print_encoded_string(Print* print, const char* str) {
568  int bytes = 0;
569  for (int i = 0; str[i] != 0; i++) {
570  if (((str[i] >= 'A') && (str[i] <= 'Z')) ||
571  ((str[i] >= 'a') && (str[i] <= 'z')) ||
572  ((str[i] >= '0') && (str[i] <= '9')) ||
573  (str[i] == '-') || (str[i] == '_') ||
574  (str[i] == '.') || (str[i] == '~')) {
575  bytes += print->print(str[i]);
576  } else {
577  // Encode all other characters
578  bytes += print->print('%');
579  bytes += print->print(HEX(str[i] / 16));
580  bytes += print->print(HEX(str[i] % 16));
581  }
582  }
583  return bytes;
584 }

598 int M2XNanodeClient::waitForString(const char* origin, int len, const char* str) {
599  int i, j, k, cmp;
600  if (str[0] == '\0') { return 0; }
601  for (i = 0; i < len; i++) {
602  for (j = i, k = 0; (j < len) && (str[k] != '\0'); j++, k++) {
603  if (_case_insensitive) {
604  cmp = tolower(origin[j]) - tolower(str[k]);
605  } else {
606  cmp = origin[j] - str[k];
607  }
608  if ((str[k] != '*') && (cmp != 0)) {
609  // No match
610  break;
611  }
612  }
613  if (str[k] == '\0') {
614  // Full match
615  return j;
616  }
617  }
618  return E_NOMATCH;
619 }
